```
// CompUnit    ::= FuncDef;
// FuncDef     ::= FuncType IDENT "(" ")" Block;
// FuncType    ::= "int";
// Block       ::= "{" {BlockItem} "}";
// BlockItem   ::= Decl | Stmt;
// Decl        ::= ConstDecl | VarDecl;
// ConstDecl   ::= "const" BType ConstDef {"," ConstDef} ";";
// BType       ::= "int";
// ConstDef    ::= IDENT "=" ConstInitVal;
// ConstInitVal::= ConstExp;
// VarDecl     ::= BType VarDef {"," VarDef} ";";
// VarDef      ::= IDENT | IDENT "=" InitVal;
// InitVal     ::= Exp;
// Stmt        ::= LVal "=" Exp ";" | "return" Exp ";";
// Exp         ::= LOrExp;
// LVal        ::= IDENT;
// PrimaryExp  ::= "(" Exp ")" | LVal | Number;
// Number      ::= INT_CONST;
// UnaryExp    ::= PrimaryExp | UnaryOp UnaryExp;
// UnaryOp     ::= "+" | "-" | "!";
// MulExp      ::= UnaryExp | MulExp ("*" | "/" | "%") UnaryExp;
// AddExp      ::= MulExp | AddExp ("+" | "-") MulExp;
// RelExp      ::= AddExp | RelExp ("<" | ">" | "<=" | ">=") AddExp;
// EqExp       ::= RelExp | EqExp ("==" | "!=") RelExp;
// LAndExp     ::= EqExp | LAndExp "&&" EqExp;
// LOrExp      ::= LAndExp | LOrExp "||" LAndExp;
// ConstExp    ::= Exp;
```
```
运算符的优先级（越往下优先级越高）：
LOrExp (逻辑或)：优先级最低。||
LAndExp (逻辑与)：&&
EqExp (相等性判断)：==, !=
RelExp (关系运算)：<, >, <=, >=
AddExp (加减法)：+, -
MulExp (乘除法)：*, /, %
UnaryExp (单目运算)：优先级最高。!, -, +
```
```
Lv4 新增语法说明：
1. Block 现在可以包含多个 BlockItem（声明或语句）
2. Decl 包括常量声明和变量声明
3. ConstDecl 定义常量，必须在编译时计算初始值
4. VarDecl 定义变量，可以有初始值也可以没有
5. Stmt 新增赋值语句：LVal "=" Exp ";"
6. LVal 用于引用变量或常量
7. PrimaryExp 现在可以是 LVal（读取变量/常量值）
```

                 │    │                   │
                 │    │                   └── Exp (表达式)
                 │    │                        │
                 │    │                        └── LOrExp (逻辑或表达式)
                 │    │                             │

                 │    │
                 │    └── VarDecl (变量声明)
